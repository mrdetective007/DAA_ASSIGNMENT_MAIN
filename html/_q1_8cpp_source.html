24bool bfs(vector<vector<Edge>>& graph, vector<int>& parent, int s, int t) {
25 int n = graph.size();

27 queue<int> q;
28 q.push(s);
29 visited[s] = true;
30 parent[s] = -1;
31 while (!q.empty()) {
32 int u = q.front();
33 q.pop();
34 visited[u] = true;
35 for (auto& e : graph[u]) {
36 int v = e.u;
37 if (!visited[v] && e.capacity > e.flow) {
38 visited[v] = true;

40 q.push(v);

42 }
43 }

46int augment(vector<vector<Edge>>& graph, vector<int>& parent, int s, int t) {
47 int n = graph.size();

49 for (int v = t; v != s; v = parent[v]) {

51 for (auto& e : graph[u]) {

53 min_flow = min(min_flow, e.capacity - e.flow);

62 e.flow += min_flow;
63 break;

68 e.flow -= min_flow;
